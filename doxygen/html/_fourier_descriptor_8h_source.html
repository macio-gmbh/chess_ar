15 Point findCenter(Mat wdw)
16 {
17 // find the center position of the hand
18  double colSum = 0.f, rowSum = 0.f, totSum = 0.f;
19  int ctrRow = 0, ctrCol = 0;
20  Point ctrPt;
21  for (int ii = 0; ii < wdw.cols; ii++)
22  {
23  for (int jj = 0; jj < wdw.rows; jj++)
24  {
25  if (wdw.at<uchar>(Point(ii, jj)) != 0)
26  {
27  colSum = colSum + ii;
28  rowSum = rowSum + jj;
29  totSum++;
30  }
31  }
32  }
33 // center of the hand in cartesian coords
34  ctrPt.x = static_cast<int>(colSum / totSum);
35  ctrPt.y = static_cast<int>(rowSum / totSum);
36  return ctrPt;
37 }

68 Mat calcFourierDescriptor(Mat wdw)
69 {
70  // Find the center position of the hand
71  Point ctrPt;
72  ctrPt = findCenter(wdw);
73 
74  // Find the contours of the silhouette
75  std::vector <std::vector<cv::Point>> contours;
76  std::vector <Vec4i> hierarchy;
77 
78  findContours(wdw, contours, hierarchy, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE);
79 
81  std::vector <std::vector<Point>> contours_poly(contours.size());
82  std::vector <Rect> boundRect(1);
83  double maxArea = 0.0f; // store the max area of all the convex hulls
84  int maxInd = 0; // store contour id of max area
85  for (int i = 0; i < contours.size(); i++)
86  {
87  approxPolyDP(Mat(contours[i]), contours_poly[i], 3, true);
88  double currArea = cv::contourArea(contours_poly[i]);
89  if (currArea > maxArea)
90  {
91  maxInd = i;
92  maxArea = currArea;
93  }
94  }
95  int lineThickness = -1;
96 // blank out the window
97  wdw(Range::all(), Range::all()) = 0;
98 // maxInd = largest contour
99  drawContours(wdw, contours, maxInd, 255, lineThickness, 8, std::vector<Vec4i>(), 0,
100  Point());
101  Mat frame;
102  wdw.copyTo(frame); // Make working copy
104 // Repeat previous steps
106  ctrPt = findCenter(frame);
107 
108 
109  // Find the contours of the silhouette
110  contours.clear();
111  hierarchy.clear();
112  findContours(frame, contours, hierarchy, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE);
113 
114  if (contours.size() == 0)
115  return Mat();
116 
117 // Remove contours that are along the bottom edge of the window
118  size_t dim1 = contours[0].size(); // number of contour points
119  int r = wdw.rows, c = wdw.cols;
120  for (int ii = 0; ii < dim1; ii++)
121  {
122  if ((contours[0][ii].x == 0) || (contours[0][ii].x == (c - 1)) ||
123  (contours[0][ii].y == 0) || (contours[0][ii].y == r - 1))
124  {
125  contours[0].erase(contours[0].begin() + ii);
126  }
127  }
128 
129 // Find equidistant points along the contour
130  int N = 16; // N point DFT
131 // size_t dim1 = contours.size();
132  size_t dim2 = contours[0].size(); // number of contour points
133  int shift = static_cast<int>(floor((dim2 / (N)) + 0.5));
134  // delta between contour samples along the contour
135  std::vector <Point> K; // vector to store the sample contour points
136  // Find and store equidistant points along the contour
137  for (int ii = 0; ii < N; ii++)
138  { // Get N samples along the contour
139  if (shift * ii > dim2)
140  {
141  break;
142  }
143  else
144  {
145  K.push_back(contours[0][shift * ii]);
146  }
147  }
148  std::vector<double> rt; // Euclidean distance vector
149  for (int ii = 0; ii < K.size(); ii++)
150  {
151 // find distance from each point to the hand center
152  K[ii] -= ctrPt;
153 // compute Euclidean distance, r(t)
154  rt.push_back(norm(K[ii]));
155  }
156 // take the magnitude of the N point FFT
157  Mat img = Mat(rt);
158  Mat planes[] = {Mat_<double>(img), Mat::zeros(img.size(), CV_64F)};
159  Mat complexI;
160  merge(planes, 2, complexI);
161  dft(complexI, complexI, DFT_COMPLEX_OUTPUT);
162  split(complexI, planes); // planes[0] = Re(DFT(I), planes[1] = Im(DFT(I))
163  magnitude(planes[0], planes[1], planes[0]); // planes[0] = magnitude
164  Mat f_all = planes[0].t();
165  // Normalize by the f0 (DC value)
166 
167  f_all /= f_all.at<double>(0);
168  // Note: This depends on the data type of planes[](BOTHpartsof the constructor)
169  // Only store the first (N/2) unique coefficient values (excludes the DC coeff, which
170  //is always 1)
171  // Ex: If N = 16, then keep 8 of the coeffs.
172  int offset = static_cast<int>(floor((N / 2) + 0.5) + 1); // offset =
173  round(N / 2) + 1;
174  // Note: Adding 1 to get all unique coeffs
175  Mat ft = f_all.colRange(Range(1, offset));
176  return ft;
177 }
